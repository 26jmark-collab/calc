\section*{Kinematics \& Graphs}
\text{Area under } F\text{-}t = \Delta p
\text{Area under } v\text{-}t = \Delta x
\text{Area under } F\text{-}x = \Delta KE
\text{Slope of } \Delta x \text{ vs } t^2 = \frac{1}{2}a
t_{stop} = \frac{2d}{v_0}
t_{stop} = \frac{v_0}{\mu g} \text{ (friction only)}
\text{Projectile speed at peak: } v = v_0\cos\theta \text{ (never zero)}

\section*{Forces}
F_{contact} = m_{behind} \cdot a
a_{system} = \frac{F_{driver}}{m_{total}}
T = \frac{m_{target} \cdot F_{driver}}{m_{total}}
F_N = m(g+a) \text{ (elevator up)}
F_N = m(g-a) \text{ (elevator down)}
F_{\parallel} = mg\sin\theta
F_{\perp} = F_N = mg\cos\theta

\section*{Circular Motion}
\sum F_{toward} = \frac{mv^2}{r}
mg - F_N = \frac{mv^2}{r} \text{ (hilltop)}
F_N - mg = \frac{mv^2}{r} \text{ (dip)}
F_N + mg = \frac{mv^2}{r} \text{ (loop top)}
\mu = \frac{v^2}{rg} \text{ (flat turn)}
T_{conical} = 2\pi\sqrt{\frac{r}{g\tan\theta}} \text{ (mass independent)}

\section*{Energy}
\Delta KE = W_{net}
\Delta KE + \Delta U_g = -W_{friction}
W_g = mgh \text{ (vertical only, path independent)}
P = \frac{W}{\Delta t} = \frac{mgh}{\Delta t}
KE_{max} = \frac{1}{2}kA^2 \text{ (SHM, mass cancels)}
KE_{total} = \frac{1}{2}mv^2 + \frac{1}{2}I\omega^2 \text{ (rolling)}
|\Delta U_2| = |\Delta U_1| + \Delta KE \text{ (Atwood)}
P = Fv \text{ (constant velocity: } F = \mu mg)

\section*{Momentum}
J = F\Delta t = \Delta p = \text{area under } F\text{-}t
v_{cm} = \frac{p_{total}}{m_{total}} = \text{constant (no external force)}
\frac{\Delta p_{system}}{\Delta t} = F_{net,external}

\section*{Rotation}
I_{end} = I_{cm} + md^2
I_{cm} = I_{end} - md^2
L = mRv \text{ (point mass on string)}
\frac{\alpha_B}{\alpha_A} = \frac{I_A}{I_B} \text{ (same torque)}
a = R\alpha
v = R\omega

\section*{Fluids}
P_1 + \rho g h_1 + \frac{1}{2}\rho v_1^2 = P_2 + \rho g h_2 + \frac{1}{2}\rho v_2^2
v = \sqrt{2g(D-d)} \text{ (open container, } P_1=P_2, v_1=0)
x = \sqrt{4d(D-d)} \text{ (horizontal distance)}
x \propto \sqrt{h} \text{ (concave down graph)}
\text{Always: } P_1=P_2 \text{ (both open to air), } v_1\approx 0 \text{ (large container)}

\section*{SHM}
\omega = \sqrt{\frac{k}{m}}
T_{spring} = 2\pi\sqrt{\frac{m}{k}}
T_{pendulum} = 2\pi\sqrt{\frac{L}{g}} \text{ (mass independent)}
KE_{max} = \frac{1}{2}kA^2 \text{ (same for any mass, same k and A)}

\section*{Gravity}
F_g = \frac{Gm_1m_2}{r^2} \text{ (decreases with distance)}
g_{planet} = \frac{GM}{r^2}
\text{Treat } g \text{ as constant only when } h \ll r_{planet}
